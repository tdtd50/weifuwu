## 📐 系统架构图

```mermaid
flowchart TB

subgraph ClientLayer["客户端层"]
    Web["Web 前端（Vue + Nginx）"]
end

subgraph GatewayLayer["网关层"]
    Gateway["API Gateway\n统一入口 / 路由转发"]
end

subgraph ServiceLayer["微服务层"]
    UserService["用户服务\nUser Service"]
    ProductService["商品服务\nProduct Service"]
    OrderService["订单服务\nOrder Service"]
end

subgraph InfraLayer["基础设施层"]
    Nacos["Nacos\n服务注册与发现"]
    MySQLUser["MySQL\n用户数据库"]
    MySQLProduct["MySQL\n商品数据库"]
    MySQLOrder["MySQL\n订单数据库"]
end

Web --> Gateway
Gateway --> UserService
Gateway --> ProductService
Gateway --> OrderService

UserService --> MySQLUser
ProductService --> MySQLProduct
OrderService --> MySQLOrder

UserService --> Nacos
ProductService --> Nacos
OrderService --> Nacos
Gateway --> Nacos
